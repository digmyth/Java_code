public class test {
    public static void main(String[] args){
        int x = (int)-8;
        System.out.println((int)(x>>1));  //算术右移， 带符号,返回 -4
    }
}



public class test {
    public static void main(String[] args){
        int x = (int)-8;
        System.out.println((int)(x>>>1));  //逻辑右移，不带符号，最高位直接补0，返回2147483644 
    }
}


#### C语言中 算术右移，最高位补符号位，即负数补1，正数补0 ##############
#include <stdio.h>

int main(){
    	int a=-8;
	printf("%d",-8>>1);  //返回-4
} 


#####  逻辑右移  ## 无符号右移就是逻辑右移，最高位直接补0  ##############
#include <stdio.h>

int main(){
    unsigned	int a=(unsigned int)-8;
    printf("%u",-8>>1);   // 4294967292
} 


结论： java/C 逻辑右移结果不一样，还不清楚
